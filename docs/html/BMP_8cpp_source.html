183void generateGaussianKernel(double& sigma, double kernel[7][7]) {
184
185 double sum = 0.0;
186 for (int i = -3; i <= 3; ++i) {
187 for (int j = -3; j <= 3; ++j) {
188 kernel[i + 3][j + 3] = (1.0 / (2 * PI * sigma * sigma)) * exp_approx(-(i * i + j * j) / (2 * sigma * sigma));
189 sum += kernel[i + 3][j + 3];
190 }
191 }
192
193 for (int i = 0; i < 7; ++i) {
194 for (int j = 0; j < 7; ++j) {
195 kernel[i][j] /= sum;
196 }
197 }
198}

208void applyGaussianFilter(unsigned char**& image, const int width, const int height, double kernel[7][7]) {
209 auto start = std::chrono::high_resolution_clock::now();
210 int kSize = 5;
211 int half = kSize / 2;
212
213 unsigned char** newImage = new unsigned char* [height];
214 for (int i = 0; i < height; ++i) {
215 newImage[i] = new unsigned char[width * 3];
216 }
217 #pragma omp parallel for collapse(2)
218 for (int i = 0; i < height; ++i) {
219 for (int j = 0; j < width; ++j) {
220 for (int c = 0; c < 3; ++c) {
221 double sum = 0.0;
222
223 for (int iBorder = -half; iBorder <= half; ++iBorder) {
224 for (int jBorder = -half; jBorder <= half; ++jBorder) {
225 int X = j + iBorder;
226 int Y = i + jBorder;
227
228 if (X >= 0 && X < width && Y >= 0 && Y < height) {
229 sum += (double)(image[Y][X * 3 + c]) * kernel[iBorder + half][jBorder + half];
230 }
231 }
232 }
233 sum = std::max(0.0, std::min(255.0, sum));
234 newImage[i][j * 3 + c] = static_cast<unsigned char>(sum);
235 }
236 }
237 }
238 for (int i = 0; i < height; ++i) {
239 for (int j = 0; j < width * 3; ++j) {
240 image[i][j] = newImage[i][j];
241 }
242 }
243
244 for (int i = 0; i < height; ++i) {
245 delete[] newImage[i];
246 }
247 delete[] newImage;
248 auto end = std::chrono::high_resolution_clock::now();
249 std::chrono::duration<double> elapsed = end - start;
250 std::cout << "Gauss computation time: " << elapsed.count() << " seconds\n";
251}

253double exp_approx(double x) {
254 double result = 1.0;
255 double term = 1.0;
256 int n = 1;
257 const double epsilon = 1e-10;
258
259
260 while (true) {
261 term *= x / n;
262 result += term;
263
264
265 if (std::abs(term) < epsilon) {
266 break;
267 }
268
269 ++n;
270 }
271
272 return result;
273}
